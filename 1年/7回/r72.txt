プログラミング説明
#include<stdio.h>
#include<math.h>
main()
{
	int i,j;
	float a[2][2];
	float theta;
	float x[2];
	float y[2];


	printf("回転（ラジアン）:");
	scanf("%f",&theta);


	a[0][0]=cos(theta);
	a[0][1]=-sin(theta);
	a[1][0]=sin(theta);
	a[1][1]=cos(theta);


	printf("元の座標:");
	scanf("%f %f",&x[0],&x[1]);

	y[0]=0;
	y[1]=0;
	for(i=0;i<=1;i++){
	for(j=0;j<=1;j++){
	
	y[i]+=x[j]*a[i][j];
	}
	}

	printf("変換後の座標(%f %f)\n",y[0],y[1]);
}

sin,cosなどの数学用の関数を用いるのでmath.hを使用する。
実数であるからそれぞれの値をfloatで宣言する
回転前の座標をx[0],x[1]で表し、回転後の座標をy[0],y[1]で表すし
回転する分の値をa[0][0]からa[1][1]の行列で表す
y[0]=x[0]*a[0][0]+x[1]*a[0][1]
y[1]=x[0]*a[1][0]+x[1]*a[1][1]
で表すことができるで
yの値とaの行の値、xの値とaの列の値が等しいことを利用し、
上のようにfor文を２回用いて表すことができる。
