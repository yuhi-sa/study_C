プログラミングの説明

#include<stdio.h>
main()
{
	int i;

	for(i=20; i>0; i--)
	{
		if(i%2==0)
		printf("%d,",i);
	}
	printf("\n");
	i=20;
	while(i>0)
	{
		if(i%2==0)
		printf("%d,",i);
		i--;
	}
	printf("\n");
	i=20;
	do{	
	if(i%2==0)
		printf("%d,",i);
		i--;
	 }	
	while(i>0);
	printf("\n");
	
	
}



for文の中の(i=20; i>0; i--)はiは0より大きい数で、20から1ずつ減らしていくという条件、その後if文で2で割り切れるという条件の時のみiがprintfで表示されるように指示
while文の中はiは0より大きいという条件で、その後if文で2で割り切れるという条件の時のみiがprintfで表示されるように指示し、iを1ずつ減らして繰り返す
do-while文の中はif文で2で割り切れるという条件の時のみiがprintfで表示されるように指示し、iを1ずつ減らして繰り返し、i>0を満たせなくなったら繰り返しをやめる
また、各文ごとにprintf("\n")で改行するように指示している
